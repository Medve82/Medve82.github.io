Kriptovaluta Árfolyamok Vizualizáció

# Python kód a vizualizációhoz import yfinance as yf import pandas as pd import plotly.graph_objects as go # Bitcoin adatok letöltése btc_data = yf.download('BTC-USD', period='5y', interval='1d') # Indexből oszloppá alakítás btc_data.reset_index(inplace=True) btc_data['Date'] = pd.to_datetime(btc_data['Date']) # Átnevezés btc_data['Price'] = btc_data['Close'] # Hover szöveg előállítása btc_data['Date_str'] = btc_data['Date'].dt.strftime('%Y-%m-%d') btc_data['Price_str'] = btc_data['Price'].map(lambda x: f"${x:,.2f}") btc_data['hover_text'] = "Dátum: " + btc_data['Date_str'] + "
Záróár: " + btc_data['Price_str'] # Interaktív vonaldiagram fig = go.Figure( go.Scatter( x=btc_data['Date'], y=btc_data['Price'], mode='lines', line=dict(color='darkorange', width=0.6), hovertext=btc_data['hover_text'], hoverinfo='text' ) ) # Layout beállítások fig.update_layout( title="Bitcoin árfolyam időbeli alakulása (Yahoo Finance)", xaxis_title="Dátum", yaxis_title="Záróár (USD)", template="simple_white", hovermode="x unified" ) # Ábra megjelenítése fig.show()
